COMANDOS GIT:

1) git clone <url>
	copia o repositório da nuvem
	
2) git add .
	adiciona as mudanças na linha do tempo
	
3) git commit -m "<mudanças feitas>"
	cria um nome para a mudanças
	
4) git push
	envia as mudanças para a nuvem
	
5) git pull
	baixa as mudanças da nuvem
	

6) git status
	mostra as mudanças locais
	
7) git fetch
	baixa as mudanças na nuvem
